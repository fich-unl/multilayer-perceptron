def constantProduct(training, y)
  shift = []
  constant = @nu/2 * (training.last - y)
  for i in 0..(@nInputs - 1)
    shift << training[i] * constant
  end
  @umbral = @umbral - constant
  shift
end

def subtractArray(shift)

def training(matrix)
  @epocas.times do
    matrix.each do |training|
      y = calculate(training)
      update(training, y)
    end
  end
